# M-30 学习成效量化与报告

---

## 一、模块定位

| 属性 | 值 |
|------|-----|
| 模块编号 | M-30 |
| 模块名称 | 学习成效量化与报告 |
| 所属分类 | 自学习 |
| 优先级 | P3 |
| 上游依赖 | M-25（学习数据收集）、M-26（偏好提炼与规则生成）、M-27（规则时效性管理）、M-29（学习成果应用） |
| 下游依赖 | 无（终端输出模块） |
| 触发方式 | 定时任务（每周一 08:00）+ 手动触发 |

### 功能概述

本模块是自学习系统的**效果评估层**。通过计算四大核心指标（人工修改率、一次通过率、平均审阅耗时、系统自动通过率），量化自学习系统的实际成效。定期生成"自学习报告"，汇总新增规则、偏好库更新、修改率趋势、高置信度TOP10规则、新偏好特征、异常项等信息，帮助运营人员全面了解自学习系统的运行状态和效果。

### 设计目标

1. **四大核心指标**：量化自学习效果，驱动持续优化
2. **趋势可视化**：展示指标变化趋势，而非仅单点数据
3. **可操作洞察**：报告包含需要人工关注的异常项
4. **定期自动生成**：减少人工报告制作成本

### 4.2 四大核心指标计算

```python
def calculate_core_metrics(start_date, end_date):
    """
    计算四大核心指标
    """
    # 获取周期内的审阅数据
    review_data = get_review_data(start_date, end_date)
    learning_records = get_learning_records(start_date, end_date)
    
    total_reviews = len(review_data)
    
    # === 1. 人工修改率 ===
    # 公式：modifications / total_reviews * 100
    # 目标：递减（说明系统生成的内容越来越符合用户偏好）
    modifications = sum(1 for r in review_data if r["review_status"] in ["modify", "reject"])
    manual_modification_rate = round(modifications / total_reviews * 100, 1) if total_reviews > 0 else 0
    
    # 按类别细分
    by_category = {}
    for category in ["copy", "cover", "video", "material", "topic", "strategy"]:
        cat_records = [r for r in learning_records if r["category"] == category]
        cat_total = len(cat_records)
        cat_mods = sum(1 for r in cat_records if r["action_type"] in ["modify", "reject"])
        by_category[category] = round(cat_mods / cat_total * 100, 1) if cat_total > 0 else 0
    
    # === 2. 一次通过率 ===
    # 公式：pass_count / total_reviews * 100
    # 目标：递增（说明首次生成的内容质量越来越高）
    pass_count = sum(1 for r in review_data if r["review_status"] == "pass")
    first_pass_rate = round(pass_count / total_reviews * 100, 1) if total_reviews > 0 else 0
    
    # 按平台细分
    by_platform = {}
    for platform in ["wechat", "douyin", "xiaohongshu", "bilibili", "weibo", "kuaishou", "zhihu"]:
        plat_reviews = [r for r in review_data if r.get("target_platform") == platform]
        plat_total = len(plat_reviews)
        plat_pass = sum(1 for r in plat_reviews if r["review_status"] == "pass")
        by_platform[platform] = round(plat_pass / plat_total * 100, 1) if plat_total > 0 else None
    
    # === 3. 平均审阅耗时 ===
    # 公式：SUM(review_duration_seconds) / COUNT(reviews)
    # 目标：递减（说明审阅越来越高效）
    durations = [r["review_duration_seconds"] for r in review_data if "review_duration_seconds" in r]
    avg_duration = round(sum(durations) / len(durations), 1) if durations else 0
    
    # 耗时分布
    distribution = {
        "under_30s": sum(1 for d in durations if d < 30),
        "30s_to_60s": sum(1 for d in durations if 30 <= d < 60),
        "60s_to_120s": sum(1 for d in durations if 60 <= d < 120),
        "over_120s": sum(1 for d in durations if d >= 120)
    }
    
    # === 4. 系统自动通过率 ===
    # 公式：auto_approved_count / total_content_count * 100
    # 目标：Mode C下递增
    auto_approved = sum(1 for r in review_data if r.get("auto_approved", False))
    auto_approval_rate = round(auto_approved / total_reviews * 100, 1) if total_reviews > 0 else 0
    
    # 按模式细分
    by_mode = {}
    for mode in ["mode_a", "mode_b", "mode_c"]:
        mode_reviews = [r for r in review_data if r.get("mode") == mode]
        mode_total = len(mode_reviews)
        mode_auto = sum(1 for r in mode_reviews if r.get("auto_approved", False))
        by_mode[mode] = round(mode_auto / mode_total * 100, 1) if mode_total > 0 else None
    
    return {
        "manual_modification_rate": {
            "current": manual_modification_rate,
            "by_category": by_category
        },
        "first_pass_rate": {
            "current": first_pass_rate,
            "by_platform": by_platform
        },
        "average_review_duration": {
            "current": avg_duration,
            "distribution": distribution
        },
        "auto_approval_rate": {
            "current": auto_approval_rate,
            "by_mode": by_mode
        }
    }
```

### 4.3 趋势分析

```python
def analyze_trend(current_metrics, previous_metrics):
    """
    对比当前和上一周期指标，分析趋势
    """
    trends = {}
    
    for metric_name in ["manual_modification_rate", "first_pass_rate", 
                         "average_review_duration", "auto_approval_rate"]:
        current = current_metrics[metric_name]["current"]
        previous = previous_metrics.get(metric_name, {}).get("current", current)
        
        change = current - previous
        
        # 判断趋势方向
        if metric_name in ["manual_modification_rate", "average_review_duration"]:
            # 这些指标越低越好
            if change < -5:
                trend = "improving"
            elif change > 5:
                trend = "declining"
            else:
                trend = "stable"
        else:
            # 这些指标越高越好
            if change > 5:
                trend = "improving"
            elif change < -5:
                trend = "declining"
            else:
                trend = "stable"
        
        trends[metric_name] = {
            "current": current,
            "previous": previous,
            "change": round(change, 1),
            "trend": trend
        }
    
    return trends
```

```python
def identify_abnormal_items(metrics, rules, learning_records):
    """
    识别需要人工关注的异常项
    """
    abnormal = {
        "rules_needing_attention": [],
        "metric_anomalies": [],
        "data_quality_issues": []
    }
    
    # 1. 规则异常
    for rule in rules:
        # 置信度快速下降（本周期下降超过30%）
        if rule.get("confidence_change_this_period", 0) < -30:
            abnormal["rules_needing_attention"].append({
                "rule_id": rule["rule_id"],
                "rule_name": rule["rule_name"],
                "issue_type": "confidence_rapid_decline",
                "severity": "warning",
                "description": f"置信度从{rule['previous_confidence']}降至{rule['confidence']}",
                "suggested_action": "检查是否因策略变更导致，考虑是否需要重置"
            })
        
        # 长期未验证（>90天）
        days_unverified = rule.get("days_since_verified", 0)
        if days_unverified > 90:
            abnormal["rules_needing_attention"].append({
                "rule_id": rule["rule_id"],
                "rule_name": rule["rule_name"],
                "issue_type": "long_unverified",
                "severity": "critical" if days_unverified > 120 else "warning",
                "description": f"已{days_unverified}天未验证",
                "suggested_action": "请确认此规则是否仍然适用，或标记为已过时"
            })
        
        # 零采纳率规则（>10次查询但0次采纳）
        if rule.get("query_count", 0) > 10 and rule.get("adoption_rate", 0) == 0:
            abnormal["rules_needing_attention"].append({
                "rule_id": rule["rule_id"],
                "rule_name": rule["rule_name"],
                "issue_type": "zero_adoption",
                "severity": "warning",
                "description": f"被查询{rule['query_count']}次但从未被采纳",
                "suggested_action": "考虑废弃此规则或调整规则内容"
            })
        
        # 频繁被覆盖（override_count > 5）
        if rule.get("statistics", {}).get("override_count", 0) > 5:
            abnormal["rules_needing_attention"].append({
                "rule_id": rule["rule_id"],
                "rule_name": rule["rule_name"],
                "issue_type": "frequent_override",
                "severity": "warning",
                "description": f"已被手动覆盖{rule['statistics']['override_count']}次",
                "suggested_action": "规则可能与实际需求不符，建议重新评估"
            })
    
    # 2. 指标异常
    # 修改率突增（>20%且与上期差异>10%）
    if metrics["manual_modification_rate"]["change"] > 10:
        abnormal["metric_anomalies"].append({
            "metric_name": "人工修改率",
            "expected_range": "应保持稳定或递减",
            "actual_value": metrics["manual_modification_rate"]["current"],
            "deviation": f"较上期上升{metrics['manual_modification_rate']['change']}%"
        })
    
    # 一次通过率突降
    if metrics["first_pass_rate"]["change"] < -10:
        abnormal["metric_anomalies"].append({
            "metric_name": "一次通过率",
            "expected_range": "应保持稳定或递增",
            "actual_value": metrics["first_pass_rate"]["current"],
            "deviation": f"较上期下降{abs(metrics['first_pass_rate']['change'])}%"
        })
    
    # 3. 数据质量问题
    unlabeled = sum(1 for r in learning_records if not r.get("tags") or len(r["tags"]) == 0)
    if unlabeled > len(learning_records) * 0.1:
        abnormal["data_quality_issues"].append({
            "issue": "未打标签的学习记录比例过高",
            "affected_count": unlabeled
        })
    
    return abnormal
```

### 9.3 质量约束

```
- 四大核心指标计算准确率 100%
- 报告生成完整性 100%（所有请求章节均需包含）
- 趋势分析正确率 100%
- 异常项检测覆盖率 >90%
- 报告生成时间 < 30秒
- 不允许在样本量 < 10 时给出确定性结论
- 不允许遗漏 critical 级别的异常项
```

### 9.4 自检指令

```
每次生成报告后，自检以下项目：
1. 检查四大指标是否全部计算（无遗漏）
2. 检查指标值是否在合理范围（0-100%或正数）
3. 检查趋势方向判断逻辑是否正确
4. 检查异常项是否包含了所有 critical 级别问题
5. 检查报告文件是否生成成功且可读
6. 检查 learning_reports 表是否写入成功
7. 对比上一期报告，检查数据连续性
8. 如发现指标异常突变（如修改率从20%跳到80%），检查数据源是否正确
9. 如发现异常，在报告 abnormal_items 中注明
```
